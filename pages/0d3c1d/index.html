int quick_pow(int a, int n){
    if(n == 0){
        return 1;
    }
    if(n&2){
        return quick_pow(a, n - 1) * a;
    }else{
        int tmp = quick_pow(a, n>>1);
        return tmp * tmp;
    }
}

typedef long long ll;
ll quick_pow(ll a, ll n)
{
    if (n == 0)
        return 1;
    else if (n % 2 == 1)
        return quick_pow(a, n - 1) * a % MOD;
    else
    {
        ll temp = quick_pow(a, n / 2) % MOD;
        return temp * temp % MOD;
    }
}

# 非递归快速幂

将n看成2进制，对于任意整数，都可以若干个x(10...)2x^{(10...)_2}x​(10...)​2​​​​的乘积表示。例如求x10x^{10}x​10​​

x10=x(1010)2=x(1000)2∗x(10)2x^{10}=x^{(1010)_2}=x^{(1000)_2}*x^{(10)_2} x​10​​=x​(1010)​2​​​​=x​(1000)​2​​​​∗x​(10)​2​​​​

我们将底数不断平方，遇到n的二进制的末位数为1的，就乘到结果里面。

//非递归快速幂
int qpow(int a, int n){
    int ans = 1;
    while(n){
        if(n&1)        //如果n的当前末位为1
            ans *= a;  //ans乘上当前的a  ans = a^2 * a^8
        a *= a;        //a自乘 a
        n >>= 1;       //n往右移一位
    }
    return ans;
}

a[b1,b2,b3,b4]=ab4∗(a[b1,b2,b3])10a^{[b_1,b_2,b_3,b_4]}=a^{b_4}*(a^{[b_1,b_2,b_3]})^{10} a​[b​1​​,b​2​​,b​3​​,b​4​​]​​=a​b​4​​​​∗(a​[b​1​​,b​2​​,b​3​​]​​)​10​​


class Solution {
public:
    int superPow(int a, vector<int>& b) {
		if(b.empty()) return 1; // base case
         int last = b.back();
         int part1 = quick_pow(a, last);
         b.pop_back();
         int part2 = quick_pow(quick_pow(a, b),10);
         return part1 * part2;
    }
    int quick_pow(int a, int n); //使用以上的快速幂来计算
};

接下来考虑数值溢出的问题，由于问题给出，需要用1337取模，根据取模运算：

(a∗b)(%k)=(Ak+B)(Ck+D)%k=(ACK2+ADk+BCk+BD)%k=(BD)%k\begin{aligned} (a*b)(\%k)&=(Ak+B)(Ck+D)\%k \\ &=(ACK^2+ADk+BCk+BD)\%k \\ &=(BD)\%k \\ \end{aligned} ​(a∗b)(%k)​​​​​​=(Ak+B)(Ck+D)%k​=(ACK​2​​+ADk+BCk+BD)%k​=(BD)%k​​

也就是说，两个数的乘积取模，可以对两个数分别取模，然后再对结果取模

(a∗b)%k=((a%k)∗(b%k))%k(a*b)\%k=((a\%k)*(b\%k))\%k (a∗b)%k=((a%k)∗(b%k))%k

使用递归的方法能使得代码简洁易懂，但是随着b的长度边长，栈会越来越深，并且不好调试。这里提供一个使用迭代的方法求解。

class Solution {
public:
    int superPow(int a, vector<int>& b) {
		int ans = 1;
        for(int i = 0; i < b.size(); i++){
            if(i != 0)
                ans = mypow(ans, 10);
           	ans = (ans%mod * mypow(a, b[i])%mod)%mod;
        }
        return ans;
    }
    int mypow(int a, int n){
        int ans = 1;
        while(n){
            if(n&1)
                ans = ((ans%mod)*(a%mod))%mod;
            a = ((a%mod)*(a%mod))%mod;
            n>>=1;
        }
        return ans;
    }
private:
    const int mod = 1337;
};
